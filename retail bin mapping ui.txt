def suggest_bins(percentage, total_bins, wall_unit_bins_per_unit=0, floor_unit_bins_per_unit=0):

    def get_excel_file(self):
        file_name, _ = QtWidgets.QFileDialog.getOpenFileName(None, "Select Excel File", "", "Excel Files (*.xlsx, *.xlsm)")
        if file_name:
            wb = xlrd.open_workbook(file_name) 
            sheet = wb.sheet_by_index(0) 
    
            num_rows = sheet.nrows  

            self.garment_hash = HashMap()
    
            for row in range(3, num_rows):
                self.garment_hash.set(int(sheet.cell_value(row, 0)), (sheet.cell_value(row, 2), sheet.cell_value(row, 1), sheet.cell_value(row, 3), sheet.cell_value(row, 4)))

            self.garment_hash.percentage_sold()
